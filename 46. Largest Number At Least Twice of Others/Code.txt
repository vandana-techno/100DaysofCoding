class Solution {
    public int dominantIndex(int[] nums) {

        int max = nums[0];
        int n = nums.length;
        int maxIndex = 0;
        for(int i=1;i<n;i++){
            if(nums[i]>max){
                max = nums[i];
                maxIndex = i;
            }
        }

        for(int i:nums){
            if(i!=max){
                if(i*2>max)
                    return -1;
            }
        }
        return maxIndex;
    }
}
